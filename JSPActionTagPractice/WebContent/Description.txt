<%--
	Action Tag
	 - XML 문법 사용
	 - 반드시 종료 태그를 작성해야 함
	 - 액션 태그 사이에 주석 작성 시 에러 발생
	 - 속성 값 부여 할 때 표현식 <%= %> 사용 가능
	 
		 * 문법
		 	<jsp:include page=".jsp 파일" flush="true | false">
		 	...
		 	</jsp>
	 
	 	<jsp:include>
	 	 - 외부 파일을 현재 파일에 포함
	 	 - page : 포함 할 JSP 경로
	 	   flush : 지정한 JSP를 실행 하기 전 출력 버퍼 비움 여부
	 	   
	 	 	* include 지시어와 비교
	 	 					지시어							액션태그
 	 			형식			<%@include file="path"%>		<jsp:include page="path />
 	 			표현식		사용불가							사용가능
	 			포함 방식		페이지 실행 전 포함페이지를 현재 페이지에		페이지 실행 중 포함페이지로 실행의 흐름이 이동한 후
	 						포함시킨 후 한번에 컴파일				포함페이지의 컴파일 결과값만 현재 페이지에 포함시킨 후 현재 페이지 컴파일
	 			변수			포함시킨 파일에서 생성한 변수 사용 가능		포함시킨 파일에서 생성한 변수 사용 불가
	 			page영역		공유됨							공유되지 않음
	 			request영역	공유됨							공유됨
	 		
	 		
	 	<jsp:forward>
	 	 - 다른 페이지로 포워딩
	 		
	 	<jsp:useBean>, <jsp:setProperty>, <jsp:getProperty>
	 	 - 자바빈즈를 생성하고 설정 및 추출
	 		
	 	<jsp:param>
	 	 - 다른 페이지로 매개변수를 전달
 --%>
